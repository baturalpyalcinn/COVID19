def country_total_plot(full_data, country):
    country_df = full_data[full_data["location"]==country]
    country_df = country_df.set_index("date")
    country_df.index = pd.to_datetime(country_df.index)
    
    country_cases = country_df["total_cases"]
    country_cases = country_cases[country_cases>100]
    country_deaths =  country_df["total_deaths"]
    country_deaths = country_deaths[country_deaths>0]
    
    f, ((ax1, ax2),(ax3, ax4)) = plt.subplots(2,2, sharex= True, sharey = False, figsize=(16,16))
    ax1.plot(country_cases.index, country_cases, label = "Total Cases")
    ax1.set_title(country + " Cumulative Cases (Linear Scale)")
    ax1.set_xlabel("Date")
    ax1.set_xticks(country_cases.index[::2])
    ax1.legend(["Total Cases"],loc="upper left")
    ax1.tick_params(axis="x", labelrotation = 90)

    ax2.plot(country_deaths.index, country_deaths, label = "Total Deaths", color="r")
    ax2.set_title(country + " Cumulative Deaths (Linear Scale)")
    ax2.set_xlabel("Date")
    ax2.set_xticks(country_deaths.index[::2])
    ax2.legend(["Total Deaths"], loc="upper left")
    ax2.tick_params(axis="x", labelrotation = 90)
    
    ax3.plot(country_cases.index, country_cases, label = "Total Cases")
    ax3.set_title(country + " Cumulative Cases (Log Scale)")
    ax3.set_xlabel("Date")
    ax3.set_xticks(country_cases.index[::2])
    ax3.legend(["Total Cases"],loc="upper left")
    ax3.tick_params(axis="x", labelrotation = 90)
    ax3.set_yscale("log")

    ax4.plot(country_deaths.index, country_deaths, label = "Total Deaths", color="r")
    ax4.set_title(country + " Cumulative Deaths (Log Scale)")
    ax4.set_xlabel("Date")
    ax4.set_xticks(country_deaths.index[::2])
    ax4.legend(["Total Deaths"], loc="upper left")
    ax4.tick_params(axis="x", labelrotation = 90)
    ax4.set_yscale("log")

def country_daily_plot(full_data, country, cum_case_lower_bound, cum_death_lower_bound,new_case_lower_bound, new_death_lower_bound ):
    country_df = full_data[full_data["location"]==country]
    country_df = country_df.set_index("date")
    country_df.index = pd.to_datetime(country_df.index)
    
    country_cases = country_df["total_cases"]
    country_cases = country_cases[country_cases>cum_case_lower_bound]
    country_deaths =  country_df["total_deaths"]
    country_deaths = country_deaths[country_deaths>cum_death_lower_bound]

    country_new_cases = country_df["new_cases"]
    country_new_cases = country_new_cases[country_new_cases>new_case_lower_bound]
    country_new_deaths = country_df["new_deaths"]
    country_new_deaths = country_new_deaths[country_new_deaths>new_death_lower_bound]
    
    f, (ax1, ax2) = plt.subplots(2,1, sharex= True, sharey = False, figsize=(16,8))
    ax1.bar(country_cases.index, country_cases, label = "Total Cases")
    ax1.bar(country_deaths.index, country_deaths, label = "Total Deaths", color="r")
    ax1.set_title(country + " Daily Cumulative Cases and Deaths")
    ax1.set_xlabel("Date")
    ax1.set_xticks(country_cases.index[::2])
    ax1.legend(["Cumulative Cases", "Cumulative Deaths"], loc="upper left")
    ax1.tick_params(axis="x", labelrotation = 90)

    ax2.bar(country_new_cases.index, country_new_cases, label = "New Cases")
    ax2.bar(country_new_deaths.index, country_new_deaths, label = "New Deaths", color="r")
    ax2.set_title(country + " Daily New Cases and Deaths")
    ax2.set_xlabel("Date")
    ax2.set_xticks(country_new_cases.index[::2])
    ax2.legend(["New Cases", "New Deaths"], loc="upper left")
    ax2.tick_params(axis="x", labelrotation = 90)

def log_scale_comparison(cases, countries):
    fig= plt.figure(figsize=(8,8))
    ax = plt.subplot(1,1,1)
    for i,country in enumerate(countries):
        first_100 = cases[country]
        first_100 = first_100[first_100>100]
        x = list(range(1,len(first_100)+1))
        ax.plot(x,first_100, label=country)
    ax.set_yscale('log')
    ax.set_title('Covid19 Cases - Log Scale')
    ax.set_xlabel("Day since 100th case")
    ax.set_ylabel("The number of cases - log scale")
    ax.legend(loc="lower right")
    ax.grid(True)

def compare_two_country_same_bar(cases, country1, country2):
    country1_df = cases[country1]
    country2_df = cases[country2]
    
    country1_df = country1_df[country1_df>100]
    country2_df = country2_df[country2_df>100]
    
    lc1 = len(country1_df)
    lc2 = len(country2_df)
    
    if(abs(lc1-lc2)>10):
        if(lc1>lc2):
            country1_df = country1_df[:(lc2+7)]
        else:
            country2_df = country2_df[:(lc1+7)]

    c1_x = list(range(1,len(country1_df)+1))
    c2_x = list(range(1,len(country2_df)+1))
    
    bar_width = 0.35
    c2_x=np.array(c2_x) + bar_width
    fig= plt.figure(figsize=(8,8))
    ax = plt.subplot(1,1,1)
    ax.barh(c1_x,country1_df, bar_width, label=country1)
    ax.barh(c2_x+bar_width,country2_df, bar_width, color="r", label=country2)
    ax.invert_yaxis()
    ax.grid()
    ax.set_title(country1 + " (Blue) vs. " + country2+ " (Red)")
    ax.set_xlabel("Total Cases")
    ax.set_ylabel("Day since the 100th case")
    #ax.legend("upper left",[country1, country2])
